const WebSocket = require("ws"); const wss = new WebSocket.Server({ port: 8080 }); let players = {}; let bullets = []; wss.on("connection", ws => { const id = Math.random().toString(36).substr(2,9); players[id] = { x: 400 + Math.random()*100, y: 300 + Math.random()*100, hp: 100, input: {} }; ws.send(JSON.stringify({ type:"init", id })); ws.on("message", msg => { const data = JSON.parse(msg); players[id].input = data; }); ws.on("close", () => delete players[id]); }); // 🔁 GAME LOOP (server authoritative) setInterval(() => { // 🎮 MOVIMIENTO for (let id in players) { let p = players[id]; let s = 5; if (p.input.w) p.y -= s; if (p.input.s) p.y += s; if (p.input.a) p.x -= s; if (p.input.d) p.x += s; // 🔫 DISPARO if (p.input.shoot) { for(let i=0;i<5;i++){ let spread = (Math.random()-0.5)*0.3; bullets.push({ x: p.x, y: p.y, dx: Math.cos(p.input.angle + spread) * 10, dy: Math.sin(p.input.angle + spread) * 10, owner: id }); } } } // 💥 BALAS + COLISIONES bullets.forEach((b,i)=>{ b.x += b.dx; b.y += b.dy; for (let id in players) { let p = players[id]; if (Math.hypot(b.x - p.x, b.y - p.y) < 20) { p.hp -= 10; bullets.splice(i,1); break; } } // eliminar si sale del mapa if (b.x < 0 || b.x > 2000 || b.y < 0 || b.y > 2000) { bullets.splice(i,1); } }); // 📡 ENVIAR ESTADO broadcast({ type:"state", players, bullets }); }, 50); function broadcast(data){ const msg = JSON.stringify(data); wss.clients.forEach(c=>c.send(msg)); } console.log("🔥 Server PRO corriendo en puerto 8080");
